import matplotlib.pyplot as plt

# Define lab types and their associated costs
labs = ['Govt (Low Cost)', 'Private (High Cost)', 'Overload', 'Unallocated']
costs = [800, 1600, 5000, 6000]  # Cost per swab
colors = ['green', 'orange', 'red', 'gray']  # Color for bars

# Define daily budget
budget = 18_000_000  # 18 million

# Calculate max number of swabs that can be processed within budget for each cost type
swabs_possible = [budget // cost for cost in costs]

# Plotting the bar chart
plt.figure(figsize=(10, 6))
bars = plt.bar(labs, swabs_possible, color=colors)
plt.title('Impact of Lab Type on Swab Testing Capacity under 18M Budget', fontsize=14)
plt.ylabel('Maximum Swabs That Can Be Tested', fontsize=12)
plt.xlabel('Lab Type / Cost Category', fontsize=12)

# Add value labels on top of bars

    height = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, height + 500, f"{height:,}", ha='center', va='bottom')

# Add gridlines for better readability
plt.grid(axis='y', linestyle='--', alpha=0.7)

# Improve layout
plt.tight_layout()

# Show the plot
plt.show()

from pulp import LpProblem, LpMinimize, LpVariable, lpSum, LpInteger, value
import pandas as pd

# === Actual Distance from Districts to Labs (from Exhibit 7) ===
labs = [
    {"name": "Lab 01", "type": "Govt", "FreeCap": 2158, "Cost": 800, "Distance": 3.31},
    {"name": "Lab 02", "type": "Govt", "FreeCap": 2750, "Cost": 800, "Distance": 1.52},
    {"name": "Lab 10", "type": "Govt", "FreeCap": 1886, "Cost": 800, "Distance": 1.99},
    {"name": "Lab 12", "type": "Private", "FreeCap": 1517, "Cost": 1600, "Distance": 11.76},
    {"name": "Lab 21", "type": "Private", "FreeCap": 1676, "Cost": 1600, "Distance": 2.59},
    {"name": "Lab 28", "type": "Private", "FreeCap": 1512, "Cost": 1600, "Distance": 9.28}
]

total_swabs = 12004
budget_limit = 18000000
overload_cost = 5000
penalty_cost = 6000
max_overload = 100
transport_cost_rate = 10  # 10 per km

# === LP Problem ===
model = LpProblem("Swab_Allocation_with_Actual_Transport", LpMinimize)

# === Decision Variables ===
x = {lab['name']: LpVariable(f"x_{lab['name']}", 0, lab['FreeCap'], cat=LpInteger) for lab in labs}
y = {lab['name']: LpVariable(f"y_{lab['name']}", 0, max_overload, cat=LpInteger) for lab in labs}
z = LpVariable("unallocated", 0, cat=LpInteger)

# === Objective Function with transport cost based on real distances ===
model += (
    lpSum(x[lab['name']] * (lab['Cost'] + lab['Distance'] * transport_cost_rate) for lab in labs) +
    lpSum(y[lab['name']] * (overload_cost + lab['Distance'] * transport_cost_rate) for lab in labs) +
    z * penalty_cost
)

# === Constraints ===
model += lpSum(x[lab['name']] + y[lab['name']] for lab in labs) + z == total_swabs
model += (
    lpSum(x[lab['name']] * (lab['Cost'] + lab['Distance'] * transport_cost_rate) for lab in labs) +
    lpSum(y[lab['name']] * (overload_cost + lab['Distance'] * transport_cost_rate) for lab in labs) +
    z * penalty_cost
) <= budget_limit

# === Solve ===
model.solve()

# === Results ===
results = []
for lab in labs:
    name = lab['name']
    reg = int(x[name].varValue)
    over = int(y[name].varValue)
    distance = lab['Distance']
    reg_cost = reg * (lab['Cost'] + distance * transport_cost_rate)
    over_cost = over * (overload_cost + distance * transport_cost_rate)
    results.append({
        "Lab": name,
        "Type": lab['type'],
        "Regular Allocated": reg,
        "Overload Allocated": over,
        "Total Allocated": reg + over,
        "Distance (km)": distance,
        "Regular Cost": reg_cost,
        "Overload Cost": over_cost,
        "Total Cost": reg_cost + over_cost
    })

results.append({
    "Lab": "",
    "Type": "",
    "Regular Allocated": 0,
    "Overload Allocated": 0,
    "Total Allocated": int(z.varValue),
    "Distance (km)": "",
    "Regular Cost": 0,
    "Overload Cost": 0,
    "Total Cost": int(z.varValue) * penalty_cost
})

df_final = pd.DataFrame(results)
print(df_final)
print(f"\n Total Swabs Allocated: {df_final['Total Allocated'].sum()}")
print(f" Total Cost: {value(model.objective):,.0f}")
